                    Command command = null;
                    if (event.equals(Event.GOT_DIAMOND)) {

                    } else if (event.equals(Event.GOT_START)) {
                        command = new GotStartCommand(/* ë°ì´í„° */);

                        command = new HitObstacleCommand(/* ë°ì´í„° */);

                    if (queue.isEmpty()) {
                        break;
                    }

                    command.execute();
                    handledCount++;
                }
            }
        }
