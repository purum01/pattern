State Pattern 적용하여 리팩토링하기

📋 실습 목표

주어진 코드는 시간대(주간/야간)에 따라 동작이 달라지는 보안 시스템입니다.
조건문을 사용한 방식을 상태 패턴(State Pattern)으로 리팩토링하세요.

조건 분기(if) 대신 상태 객체에 행동을 위임하고,
시각에 따라 자동으로 상태(State)를 변경하도록 구현하세요.

🛠️ 실습 조건

  1. 모든 상태는 State 인터페이스를 구현해야 합니다.
  2. 상태에 따라 버튼 행동이 다르게 동작해야 합니다.
  3. 시간이 09~17시가 아니면 야간 상태로 전환합니다.
상태 시간 변화 금고 사용 비상벨 일반 통화
주간 09~17시 유지, 이외는 야간 전환 기록 로그 보안센터 호출 보안센터 호출
야간 09~17시 진입 시 주간 전환 비상 로그 기록 보안센터 호출 통화 녹음

🛠 리팩토링 지침

  • State 인터페이스를 정의하고 행동을 선언합니다.
  • DayState, NightState 클래스를 구현하여 시간에 따라 다르게 행동하도록 합니다.
  • Context 인터페이스를 통해 상태 변경 및 기록/출력을 처리하도록 합니다.
  • SafeFrame 클래스는 Context 역할로 현재 상태에게 요청을 위임합니다.

🗂 제공 파일 (리팩토링 전)

아래 코드에서 시간 조건에 따른 if 분기들을 제거하고, 상태 패턴을 적용하세요.

SafeFrame.java


package test.state;

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class SafeFrame extends JFrame implements ActionListener {
    private JTextField textClock = new JTextField(60);
    private JTextArea textScreen = new JTextArea(10, 60);
    private JButton buttonUse = new JButton("금고사용");
    private JButton buttonAlarm = new JButton("비상벨");
    private JButton buttonPhone = new JButton("일반 통화");
    private JButton buttonExit = new JButton("종료");

    private int hour = 0;

    public SafeFrame(String title) {
        super(title);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setBackground(Color.lightGray);
        setLayout(new BorderLayout());

        // 폰트 지정 (한글 깨짐 방지)
        Font font = new Font("맑은 고딕", Font.PLAIN, 14);
        textClock.setFont(font);
        textScreen.setFont(font);
        buttonUse.setFont(font);
        buttonAlarm.setFont(font);
        buttonPhone.setFont(font);
        buttonExit.setFont(font);

        // 텍스트 필드 상단
        textClock.setEditable(false);
        add(textClock, BorderLayout.NORTH);

        // 텍스트 에어리어 중앙
        textScreen.setEditable(false);
        add(new JScrollPane(textScreen), BorderLayout.CENTER);

        // 버튼 패널 하단
        JPanel panel = new JPanel();
        panel.add(buttonUse);
        panel.add(buttonAlarm);
        panel.add(buttonPhone);
        panel.add(buttonExit);
        add(panel, BorderLayout.SOUTH);

        // 이벤트 등록
        buttonUse.addActionListener(this);
        buttonAlarm.addActionListener(this);
        buttonPhone.addActionListener(this);
        buttonExit.addActionListener(this);

        pack();
        setLocationRelativeTo(null); // 화면 중앙 배치
        setVisible(true);
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        if (e.getSource() == buttonUse) {
            if (hour >= 9 && hour < 17) {
                recordLog("주간 금고 사용");
            } else {
                recordLog("비상:야간의금고사용");
            }
        } else if (e.getSource() == buttonAlarm) {
            if (hour >= 9 && hour < 17) {
                callSecurityCenter("주간 비상벨");
            } else {
                callSecurityCenter("비상벨(야간)");
            }
        } else if (e.getSource() == buttonPhone) {
            if (hour >= 9 && hour < 17) {
                callSecurityCenter("주간 일반 통화");
            } else {
                recordLog("야간의통화녹음");
            }
        } else if (e.getSource() == buttonExit) {
            System.exit(0);
        }
    }

    public void setClock(int hour) {
        this.hour = hour;
        String clockstring = "현재 시각은 ";
        if (hour < 10) {
            clockstring += "0" + hour + ":00";
        } else {
            clockstring += hour + ":00";
        }
        System.out.println(clockstring);
        textClock.setText(clockstring);
    }

    public void callSecurityCenter(String msg) {
        textScreen.append("call! " + msg + "\n");
    }

    public void recordLog(String msg) {
        textScreen.append("record ... " + msg + "\n");
    }

}

Main.java


package state;

public class Main {
    public static void main(String[] args) {
        SafeFrame frame = new SafeFrame("State Pattern 실습용");
        while (true) {
            for (int hour = 0; hour < 24; hour++) {
                frame.setClock(hour);
                try {
                    Thread.sleep(1000); // 1초마다 시간 변경
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        }
    }
}
