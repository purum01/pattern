MySingleton.java


package singleton;

public class MySingleton {
    private static MySingleton me = new MySingleton();

    private MySingleton() {
        try {
            Thread.sleep(5000);
        } catch (InterruptedException e) {
        }
        System.out.println("MySingleton()");
    }

    public static MySingleton getInstance() {
        if (me == null) {
            me = new MySingleton();
        }
        return me;
    }
}

Main.java


package singleton;

public class Main {
    public static void main(String[] args) {
        // MySingleton my = new MySingleton(); // 생성자 호출 불가하다.
        MySingleton my = MySingleton.getInstance();
        for (int i = 0; i < 5; i++) {
            my = MySingleton.getInstance();
        }
    }
}
