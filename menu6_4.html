FramePrinter.java


package decorator;

public class FramePrinter {
    private String text;

    public FramePrinter(String text) {
        this.text = text;
    }

    public void printWithDecoration() {
        String line = "+-------------+";
        System.out.println("~~~~~~~~~~~~~~~");
        System.out.println(line);
        System.out.println("|#" + text + "#|");
        System.out.println(line);
        System.out.println("~~~~~~~~~~~~~~~");
    }
}

package decorator;

public class Main {
    public static void main(String[] args) {
        FramePrinter printer = new FramePrinter("Hello, world.");
        printer.printWithDecoration();
    }
