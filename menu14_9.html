Product.java


public abstract class Product {
    public abstract void use();
}

Factory.java


public abstract class Factory {
	public final Product create(String info) {
		Product p = createProduct(info);
		registerProduct(p);
		return p;
	}
	protected abstract Product createProduct(String info);
	protected abstract void registerProduct(Product product);
}

IDCard.java


public class IDCard extends Product {
    private String owner;

    public IDCard(String owner) {
        this.owner = owner;
        System.out.println("IDCard를 만듭니다: " + owner);
    }

    @Override
    public void use() {
        System.out.println("IDCard를 사용합니다: " + owner);
    }

    public String getOwner() {
        return owner;
    }
}

IDCardFactory.java


public class IDCardFactory extends Factory {
	private List owners = new ArrayList<>();

	@Override
	protected Product createProduct(String owner) {
		return new IDCard(owner);
	}

	@Override
	protected void registerProduct(Product product) {
		IDCard card = (IDCard) product;
		owners.add(card.getOwner());
	}

	public List getOwners() {
		return owners;
	}
}

Membership.java


public class Membership extends Product {
	private String grade;

	public Membership(String grade) {
		this.grade = grade;
		System.out.println("Membership 생성됨: 등급 " + grade);
	}

	@Override
	public void use() {
		System.out.println("Membership 사용: 등급 " + grade);
	}

	public String getGrade() {
		return grade;
	}
}

MembershipFactory.java


ublic class MembershipFactory extends Factory {
    private List grades = new ArrayList<>();

    @Override
    protected Product createProduct(String grade) {
        return new Membership(grade);
    }

    @Override
    protected void registerProduct(Product product) {
        Membership m = (Membership) product;
        grades.add(m.getGrade());
    }

    public List getGrades() {
        return grades;
    }
}

public class Main {
